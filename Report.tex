% \mfile{DisplayName}{path/to/file.m}
%   Reads and typesets a MATLAB source file with a labelled caption.
\newcommand{\mfile}[2]{%
  \vspace{0.3cm}%
  \lstinputlisting[style=mstyle, caption={\texttt{#1}}, label={lst:#1}]{#2}%
  \vspace{0.4cm}%
}

% \figfile{path/to/image}{Caption text}
%   Inserts a single centred figure at 0.82 linewidth.
\newcommand{\figfile}[2]{%
  \begin{figure}[H]
    \centering
    \includegraphics[width=0.82\linewidth]{#1}
    \caption{#2}
  \end{figure}%
}

% \imggroup{input path}{output prefix}{extension}
%   Groups Input / Visualization / Trade-off side by side for one test case.
%     #1 = full path to the input image   e.g. ./assets/input/01_landscape.jpg
%     #2 = output path prefix (no ext, no _visualization/_tradeoff suffix)
%          e.g. ./assets/output/01_landscape
%     #3 = file extension without dot     e.g. jpg
\newcommand{\imggroup}[3]{%
  \begin{figure}[H]
    \centering
    \begin{subfigure}[t]{0.32\linewidth}
      \centering
      \includegraphics[width=\linewidth]{#1}
      \caption{Input}
    \end{subfigure}\hfill
    \begin{subfigure}[t]{0.32\linewidth}
      \centering
      \includegraphics[width=\linewidth]{#2_visualization.#3}
      \caption{Visualization}
    \end{subfigure}\hfill
    \begin{subfigure}[t]{0.32\linewidth}
      \centering
      \includegraphics[width=\linewidth]{#2_tradeoff.#3}
      \caption{Trade-off}
    \end{subfigure}
    \caption{Results -- \texttt{#1}}
  \end{figure}%
  \vspace{0.3cm}%
}
